DOUBLE LINKED LIST

#include<sdio.h>
#include<stdlib.h>
typedef struct Node {
    int data;
    struct Node* prev;
    struct Node* next;
} Node;

Node* createNode(int data) {
    Node* newNode = (Node*)malloc(sizeof(Node));
    newNode->data = data;
    newNode->prev = NULL;
    newNode->next = NULL;
    return newNode;
}

intmain() {
    // Create a double linked list
    Node* head = createNode(1);
    Node* second = createNode(2);
    Node* third = createNode(3);

    // Connect the nodes
    head->next = second;
    second->prev = head;
    second->next = third;
    third->prev = second;

    // Traverse the list forward
    Node* current = head;
    while (current != NULL) {
        printf("%d ", current->data);
        current = current->next;
    }
    printf("\n");

    // Traverse the list backward
    current = third;
    while (current != NULL) {
        printf("%d ", current->data);
        current = current->prev;
    }
    printf("\n");

    return 0;
}

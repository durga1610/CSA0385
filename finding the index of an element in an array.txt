#include <stdio.h>
int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int search_element = 30;
    int index = -1;
    int i = 0;
    while (i < 5) {
        if (arr[i] == search_element) {
            index = i;
            i = 5; 
        }
        i++;
    }
    if (index != -1) {
        printf("Element found at index: %d", index);
    } else {
        printf("Element not found in the array");
    }
    return 0;
}

#include<stdio.h>
int main()
{
int arr[]={2,6,4,9,1,34,22,66,75};

 if(arr[i]==search)

printf("elemrnt found at index %d",i);
            found=1;
            break;
}
}
 if(!found)
    {
 printf("element not found");
    }
return 0;
}
